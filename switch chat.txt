def compute_momentum(df):
    df['returns'] = np.log(df['Close'] / df['Close'].shift(1))
    
    momentum = df['Close'].iloc[-1] / df['Close'].iloc[-6] - 1

from concurrent.futures import ThreadPoolExecutor
import yfinance as yf

def fetch_ticker(ticker):
    try:
        df = yf.download(ticker, period="10d", interval="1d", progress=False)
        if len(df) < 6:
            return None
        score = compute_momentum(df)
        return {"ticker": ticker, "score": score}
    except:
        return None

def get_rankings(tickers):
    results = []
    with ThreadPoolExecutor(max_workers=10) as executor:
        data = executor.map(fetch_ticker, tickers)
    
    for r in data:
        if r:
            results.append(r)
    
    return sorted(results, key=lambda x: x["score"], reverse=True)

import numpy as np

def compute_momentum(df):
    # Safety check
    if len(df) < 6:
        return None

    # 1. Log returns
    df['log_returns'] = np.log(df['Close'] / df['Close'].shift(1))

    # 2. Momentum (5 trading days ≈ 1 week)
    momentum = (df['Close'].iloc[-1] / df['Close'].iloc[-6]) - 1

    # 3. Volatility (annualized)
    volatility = df['log_returns'].std() * np.sqrt(252)

    # 4. Avoid division errors
    if volatility == 0 or np.isnan(volatility):
        return None

    # 5. Risk-adjusted score
    score = momentum / volatility

    return {
        "score": score,
        "momentum": momentum,
        "volatility": volatility
    }
🔍 What Each Line Really Means
✅ log_returns
Measures daily movement stability

✅ momentum
(df['Close'].iloc[-1] / df['Close'].iloc[-6]) - 1
→ 1-week return

✅ volatility
std * sqrt(252)
Why 252?

~252 trading days/year

Converts daily volatility → yearly scale

✅ score
momentum / volatility
This is the magic:

High return + low volatility → 🔥 high score

High return + high volatility → 😬 lower score

A	+6%	10%	0.60
B	+6%	30%	0.20
👉 Your old system: tie
👉 New system: A wins (correctly)

import numpy as np
import yfinance as yf
from fastapi import FastAPI

app = FastAPI()

def compute_momentum(df):
if df is None or df.empty or len(df) < 6:
return None

df = df.copy()

df["log_returns"] = np.log(df["Close"] / df["Close"].shift(1))

momentum = (df["Close"].iloc[-1] / df["Close"].iloc[-6]) - 1
volatility = df["log_returns"].std() * np.sqrt(252)

if volatility == 0 or np.isnan(volatility):
    return None

score = momentum / volatility

return {
    "score": float(score),
    "momentum": float(momentum),
    "volatility": float(volatility),
}
TICKERS = ["AAPL", "MSFT", "NVDA", "AMZN", "META", "AMD", "TSLA"]

def analyze_ticker(ticker):
try:
df = yf.download(ticker, period="10d", interval="1d", progress=False)

    result = compute_momentum(df)
    if result is None:
        return None

    return {
        "ticker": ticker,
        **result
    }

except Exception:
    return None

import numpy as np
import yfinance as yf
from fastapi import FastAPI

app = FastAPI()

def compute_momentum(df):
if df is None or df.empty or len(df) < 6:
return None

df = df.copy()

df["log_returns"] = np.log(df["Close"] / df["Close"].shift(1))

momentum = (df["Close"].iloc[-1] / df["Close"].iloc[-6]) - 1
volatility = df["log_returns"].std() * np.sqrt(252)

if volatility == 0 or np.isnan(volatility):
    return None

score = momentum / volatility

return {
    "score": float(score),
    "momentum": float(momentum),
    "volatility": float(volatility),
}
TICKERS = ["AAPL", "MSFT", "NVDA", "AMZN", "META", "AMD", "TSLA"]

    result = compute_momentum(df)
    if result is None:
        return None

    return {

    }

except Exception:
    return None

def compute_momentum(df):
    if df is None or df.empty or len(df) < 6:
        return None

    df = df.copy()

    df["log_returns"] = np.log(df["Close"] / df["Close"].shift(1))

    momentum = (df["Close"].iloc[-1] / df["Close"].iloc[-6]) - 1
    volatility = df["log_returns"].std() * np.sqrt(252)

    if volatility == 0 or np.isnan(volatility):
